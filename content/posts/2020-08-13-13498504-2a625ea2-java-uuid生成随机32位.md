import java.util.UUID;

/**
 * @ClassName：UuidUtils
 * @Description：uuid工具类
 * @Author：chenyb
 * @Date：2020/8/13 12:52 下午
 * @Versiion：1.0
 */
public class UuidUtils {
    /**
     * 生成uuid32位
     * @return
     */
    public static String getUUID32(){
        return UUID.randomUUID().toString().replace("-", "").toLowerCase();
    }
}
