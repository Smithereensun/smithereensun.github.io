@Test

    int a=9;

    DecimalFormat df=new DecimalFormat("0.00");

    System.out.println(df.format((float)a/b));
    System.out.println(df.format(a/(float)b));
    System.out.println(df.format((float)a/(float)b));
    System.out.println(df.format((float)(a/b)));
}
