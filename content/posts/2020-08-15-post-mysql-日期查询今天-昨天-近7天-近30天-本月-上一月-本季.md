<h1 align="left">今天</h1>
<p align="left">select&nbsp;*&nbsp;from&nbsp;表名&nbsp;where&nbsp;to_days(时间字段名)&nbsp;=&nbsp;to_days(now());</p>
<h1 align="left">昨天</h1>
<p align="left">SELECT&nbsp;*&nbsp;FROM&nbsp;表名&nbsp;WHERE&nbsp;TO_DAYS( NOW( ) )&nbsp;-&nbsp;TO_DAYS( 时间字段名)&nbsp;&lt;=&nbsp;1</p>
<h1 align="left">近7天</h1>
<p align="left">SELECT&nbsp;*&nbsp;FROM&nbsp;表名&nbsp;where&nbsp;DATE_SUB(CURDATE(), INTERVAL&nbsp;7&nbsp;DAY)&nbsp;&lt;=&nbsp;date(时间字段名)</p>
<h1 align="left">近30天</h1>
<p align="left">SELECT&nbsp;*&nbsp;FROM&nbsp;表名&nbsp;where&nbsp;DATE_SUB(CURDATE(), INTERVAL&nbsp;30&nbsp;DAY)&nbsp;&lt;=&nbsp;date(时间字段名)</p>
<h1 align="left">本月</h1>
<p align="left">SELECT&nbsp;*&nbsp;FROM&nbsp;表名&nbsp;WHERE&nbsp;DATE_FORMAT( 时间字段名,&nbsp;'%Y%m'&nbsp;)&nbsp;=&nbsp;DATE_FORMAT( CURDATE( ) ,&nbsp;'%Y%m'&nbsp;)</p>
<h1 align="left">上一月</h1>
<p align="left">SELECT&nbsp;*&nbsp;FROM&nbsp;表名&nbsp;WHERE&nbsp;PERIOD_DIFF( date_format( now( ) ,&nbsp;'%Y%m'&nbsp;) ,date_format( 时间字段名,&nbsp;'%Y%m'&nbsp;) )&nbsp;=1</p>
<h1 align="left">查询本季度数据</h1>
<p align="left">select&nbsp;*&nbsp;from&nbsp;`ht_invoice_information`&nbsp;where&nbsp;QUARTER(create_date)=QUARTER(now());</p>
<h1 align="left">查询上季度数据</h1>
<p align="left">select&nbsp;*&nbsp;from&nbsp;`ht_invoice_information`&nbsp;where&nbsp;QUARTER(create_date)=QUARTER(DATE_SUB(now(),interval&nbsp;1&nbsp;QUARTER));</p>
<h1 align="left">查询本年数据</h1>
<p align="left">select&nbsp;*&nbsp;from&nbsp;`ht_invoice_information`&nbsp;where&nbsp;YEAR(create_date)=YEAR(NOW());</p>
<h1 align="left">查询上年数据</h1>
<p align="left">select&nbsp;*&nbsp;from&nbsp;`ht_invoice_information`&nbsp;where&nbsp;year(create_date)=year(date_sub(now(),interval&nbsp;1&nbsp;year));</p>
<h1 align="left">查询当前这周的数据</h1>
<p align="left">SELECT&nbsp;name,submittime&nbsp;FROMenterprise&nbsp;WHERE&nbsp;YEARWEEK(date_format(submittime,'%Y-%m-%d'))&nbsp;=&nbsp;YEARWEEK(now());</p>
<h1 align="left">查询上周的数据</h1>
<p align="left">SELECT&nbsp;name,submittime&nbsp;FROMenterprise&nbsp;WHERE&nbsp;YEARWEEK(date_format(submittime,'%Y-%m-%d'))&nbsp;=&nbsp;YEARWEEK(now())-1;</p>
<h1 align="left">查询上个月的数据</h1>
<p align="left">&nbsp;</p>
<p align="left">select&nbsp;name,submittime&nbsp;fromenterprise&nbsp;where&nbsp;date_format(submittime,'%Y-%m')=date_format(DATE_SUB(curdate(), INTERVAL&nbsp;1&nbsp;MONTH),'%Y-%m')</p>
<p align="left">&nbsp;</p>
<p align="left">select&nbsp;*&nbsp;from&nbsp;user&nbsp;where&nbsp;DATE_FORMAT(pudate,'%Y%m')&nbsp;=&nbsp;DATE_FORMAT(CURDATE(),'%Y%m') ;</p>
<p align="left">&nbsp;</p>
<p align="left">select&nbsp;*&nbsp;from&nbsp;user&nbsp;where&nbsp;WEEKOFYEAR(FROM_UNIXTIME(pudate,'%y-%m-%d'))&nbsp;=&nbsp;WEEKOFYEAR(now())</p>
<p align="left">&nbsp;</p>
<p align="left">select&nbsp;*&nbsp;from&nbsp;user&nbsp;where&nbsp;MONTH(FROM_UNIXTIME(pudate,'%y-%m-%d'))&nbsp;=&nbsp;MONTH(now())</p>
<p align="left">&nbsp;</p>
<p align="left">select&nbsp;*&nbsp;from&nbsp;user&nbsp;where&nbsp;YEAR(FROM_UNIXTIME(pudate,'%y-%m-%d'))&nbsp;=&nbsp;YEAR(now())&nbsp;and&nbsp;MONTH(FROM_UNIXTIME(pudate,'%y-%m-%d'))&nbsp;=&nbsp;MONTH(now())&nbsp;<br><br></p>
<p align="left">select&nbsp;*&nbsp;from&nbsp;user&nbsp;where&nbsp;pudate&nbsp;between&nbsp; 上月最后一天&nbsp;and&nbsp;下月第一天</p>

<h1 align="left">查询当前月份的数据&nbsp;</h1>
<p align="left">select&nbsp;name,submittime&nbsp;fromenterprise&nbsp;&nbsp;&nbsp;wheredate_format(submittime,'%Y-%m')=date_format(now(),'%Y-%m')</p>
<h1 align="left">查询距离当前现在6个月的数据</h1>
<p align="left">select&nbsp;name,submittime&nbsp;fromenterprise&nbsp;where&nbsp;submittime&nbsp;betweendate_sub(now(),interval&nbsp;6&nbsp;month)&nbsp;and&nbsp;now();</p>
<h1 align="left">查询当前日期类型</h1>
<h2 align="left">本年份</h2>
<p align="left">SELECT DATE_FORMAT(NOW(), '%Y');</p>
<h2>本月份(数字)</h2>
<p>SELECT DATE_FORMAT(NOW(), '%m');</p>
<h2>本月份(英文)</h2>
<p>SELECT DATE_FORMAT(NOW(), '%M');</p>
<h2>本日期(数字)</h2>
<p>SELECT DATE_FORMAT(NOW(), '%d');</p>
<h2 align="left"><strong>本日期（显示数字和字母，如：16th）</strong></h2>
<p align="left">SELECT DATE_FORMAT(NOW(), '%D');</p>
<h2 align="left"><strong>当前时间（11:54:23）</strong></h2>
<p align="left">SELECT DATE_FORMAT(NOW(), '%T');</p>
<h2 align="left"><strong>当前时间（11:54:23）</strong></h2>
<p align="left">select DATE_FORMAT(now(),'%Y-%m-%d %h:%i:%s') from dual;</p>
